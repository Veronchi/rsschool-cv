function countPositivesSumNegatives(input) {
  if (input === null || input.length === 0) {
      return []
  }

  const posArr = input.filter(i => i > 0);
  const negSum = input.filter(i => i < 0)
    .reduce((prev, curr) => prev + curr, 0);

  return [posArr.length, negSum];
}
